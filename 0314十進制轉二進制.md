#include<iostream>  
using namespace std;  
  
int main(){  
    int n=0;    
    int bin[8];//用binary二進制取8位元    
    cin>>n;//輸入想要轉換的十進制數    
    if (n<0)    
    {    
        n+=256;  
    }    
    for (int i = 7; i >= 0; i--)//迴圈除以2直到餘數不能再除以2    
    {    
        bin[i]=n%2;    
        n/=2;    
    }    
    for (int i = 0; i < 8; i++)    
    {    
        cout<<bin[i];    
    }    
    cout<<"\n"; 
	return 0;   
    }
